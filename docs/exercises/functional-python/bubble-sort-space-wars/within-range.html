from main import Ship, filter_by_distance


def test_filter_one_ship_by_distance():
    s = Ship(0, 0, 0, 0)

    assert filter_by_distance([s], 50) == [s]


def test_filter_ships_by_distance():

    s1 = Ship(1, 0, 0)
    s2 = Ship(2, 0, 0)
    s3 = Ship(3, 0, 0)

    ships = [s1, s2, s3]
    assert filter_by_distance(ships, 2) == [s1, s2]
