Tests¶

from main import sum_even


def test_sum_even():
    assert sum_even([1, 1, 1, 2]) == 2
    assert sum_even([2, 1, 1, 2]) == 4
    assert sum_even([4, 1, 1, 21]) == 4
    assert sum_even([]) == 0
    assert sum_even([1, 3, 5, 7, 9]) == 0
    assert sum_even([-2]) == -2
