def repeat_front(string: str, n: int) -> str:
    new_string = ""
    slice_size = n
    while slice_size > 0:
        new_string += string[0:slice_size]
        slice_size -= 1

    return new_string

def double_char(string: str) -> str:

    new_string = ""
    for c in string:
        new_string += c + c

    return new_string

def cat_dog(string: str) -> bool:
    cat_count = 0
    dog_count = 0

    i = 0
    while i < len(string) - 2:
        substring = string[i:i+3]
        if substring == "cat":
            cat_count += 1
        elif substring == "dog":
            dog_count += 1
        i += 1

    return cat_count == dog_count

def end_other(a: str, b: str) -> bool:
    a = a.lower()
    b = b.lower()

    if a[-len(b):] == b:
        return True
    elif b[-len(a):] == a:
        return True
    else:
        return False

def xyz_there(string: str) -> bool:

        if string[i] == ".":
            i += 2
        else:
            if string[i:i+3] == "xyz":

            i += 1

    return False

def xy_balance(string: str) -> bool:
    found_y = False
    i = len(string) - 1
    while i >= 0:
        if string[i] == "y":
            found_y = True
        elif string[i] == "x":
            if not found_y:
                return False
            break
        i -= 1

    return True

def mix_string(a: str, b: str) -> str:
    new_string = ""
    i = 0
    while i < len(a) and i < len(b):
        new_string += a[i] + b[i]
        i += 1

    new_string += a[i:]
    new_string += b[i:]

    return new_string

def repeat_end(string: str, n: int) -> str:
    new_string = ""
    i = 0
    while i < n:
        new_string += string[-n:]
        i += 1

    return new_string
